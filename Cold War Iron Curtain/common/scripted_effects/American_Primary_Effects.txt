        end = canidate_listing_array_percentage^num
        value = v
        randomize_variable = {
            var = Margin_of_Error
            distribution = uniform
            min = -4
            max = 4
        }
        add_to_variable = {
            canidate_listing_array_percentage^v = Margin_of_Error
        }
    }
    USA = {
        generate_campaign_map_numbers_dem = yes
    }
    set_variable = {
        CurrentDemStateWinner = canidate_listing_array^0

        limit = {
            has_country_flag = Dem_Primary_Reforms

    }
    else = {

American_Presidential_Approval_Calculation = {
    if = {
        limit = {
            is_power_balance_side_active = {
                id = USA_approval_balance
                side = USA_disapproval_left_side_title
            }
        }
        set_variable = {
            Presidential_Approval_Base = power_balance_value
        }
        divide_variable = {
            Presidential_Approval_Base = 2
        }
        set_variable = {
            Presidential_Approval = 0.50
        }
        subtract_from_variable = {
            Presidential_Approval = Presidential_Approval_Base
        }
    }
    else = {
        set_variable = {
            Presidential_Approval_Base = power_balance_value
        }
        divide_variable = {
            Presidential_Approval_Base = 2
        }
        set_variable = {
            Presidential_Approval = 0.50
        }
        add_to_variable = {
            Presidential_Approval = Presidential_Approval_Base
        }
    }
